next = null;
        bottom = null;
    }
}
*/
class Solution {
    public Node flatten(Node root) {
        // Base case: empty or single node with no bottom
        if (root == null || (root.next == null && root.bottom == null)) {
            return root;
        }
        
        // Step 1: Recursively flatten the next (right) part
        Node rightFlattened = flatten(root.next);
        
        // Step 2: Merge current bottom list with the flattened right part
        Node merged = mergeTwoLists(root, rightFlattened);
        
        // Step 3: Set next to null (flattened list uses bottom only)
        root.next = null;
        
        return merged;
    }
    
    // Helper: Merge two sorted linked lists (using bottom pointer)
    private Node mergeTwoLists(Node a, Node b) {
        // Use a dummy node to build result
        Node dummy = new Node(0);
        Node tail = dummy;
        
        while (a != null && b != null) {
            if (a.data <= b.data) {
                tail.bottom = a;
                a = a.bottom;
            } else {
                tail.bottom = b;
                b = b.bottom;
            }
            tail = tail.bottom;
